**2. Extend `src/layouts/BaseLayout.astro` AnimationController with counter logic.**

Inside the existing `<script is:inline>` IIFE, add counter logic. The counter must fire INSIDE the `if (entry.isIntersecting)` block, after `entry.target.classList.add('is-visible')` and before `observer.unobserve(entry.target)`.

Replace the entire `<script is:inline>` block (lines 36-60) with:

```html
<script is:inline>
  (function () {
    document.documentElement.classList.add('js-enabled');
    var reducedMotion = window.matchMedia('(prefers-reduced-motion: reduce)').matches;

    // Pre-set counter values for reduced-motion users (D-12)
    if (reducedMotion) {
      document.querySelectorAll('[data-counter-target]').forEach(function(counter) {
        var target = parseInt(counter.getAttribute('data-counter-target'), 10);
        var suffix = counter.getAttribute('data-counter-suffix') || '';
        counter.textContent = target.toLocaleString() + suffix;
      });
    }

    var observer = new IntersectionObserver(
      function (entries) {
        entries.forEach(function (entry) {
          if (entry.isIntersecting) {
            entry.target.classList.add('is-visible');

            // Phase 8: counter animation — fire count-up on child elements with data-counter-target
            if (!reducedMotion) {
              var counterEls = entry.target.querySelectorAll('[data-counter-target]');
              counterEls.forEach(function(counter) {
                var target = parseInt(counter.getAttribute('data-counter-target'), 10);
                var suffix = counter.getAttribute('data-counter-suffix') || '';
                var duration = 400;
                var start = null;
                function step(timestamp) {
                  if (!start) start = timestamp;
                  var progress = Math.min((timestamp - start) / duration, 1);
                  counter.textContent = Math.floor(progress * target).toLocaleString() + suffix;
                  if (progress < 1) {
                    requestAnimationFrame(step);
                  } else {
                    counter.textContent = target.toLocaleString() + suffix;
                  }
                }
                requestAnimationFrame(step);
              });
            }

            observer.unobserve(entry.target);
          }
        });
      },
      { threshold: 0.15 }
    );
    document.querySelectorAll('[data-animate]').forEach(function (el) {
      // ANIM-03 stagger: read data-animate-delay, set as inline animationDelay
      var delay = el.getAttribute('data-animate-delay');
      if (delay) {
        el.style.animationDelay = delay;
      }
      observer.observe(el);
    });
  })();
</script>
```

Key points:
- Uses `var` declarations (not const/let) to match existing is:inline script convention
- `reducedMotion` check at the top: immediately sets final values if reduce-motion is on (D-12)
- Counter logic is inside the `if (entry.isIntersecting)` block, so it only fires when scrolled into view (Pitfall 2)
- Uses `requestAnimationFrame` (not setInterval) for smooth animation on 120Hz displays
- Duration is 400ms (within D-12 range of 300-500ms)
- `toLocaleString()` adds comma formatting (1200 -> "1,200")
- The `observer.unobserve` is AFTER counter logic, ensuring the counter fires exactly once
