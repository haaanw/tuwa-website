**c) getAngle function** — pointer angle relative to wheel center with viewBox scaling per RESEARCH.md Pitfall 1:
```javascript
function getAngle(e) {
  var rect = svgEl.getBoundingClientRect();
  var scaleX = 480 / rect.width;
  var scaleY = 480 / rect.height;
  var dx = (e.clientX - rect.left) * scaleX - CX;
  var dy = (e.clientY - rect.top) * scaleY - CY;
  return Math.atan2(dy, dx) * (180 / Math.PI);
}
```

**d) Pointer Events drag handlers:**

`pointerdown` on svgEl:
- Set isDragging = true
- Call svgEl.setPointerCapture(e.pointerId) — per RESEARCH.md for out-of-bounds drag
- Record startAngle = getAngle(e) - currentRotation
- Reset velocitySamples = [] and dragDistance = 0
- Cancel active inertia: if (rafId) cancelAnimationFrame(rafId)
- Remove snap transition: groupEl.classList.remove('wheel-snap-transition')

`pointermove` on svgEl:
- if (!isDragging) return
- Compute angle = getAngle(e)
- Compute newRotation = angle - startAngle
- Normalize delta (RESEARCH.md Pitfall 4 — angle discontinuity at +/-180):
  ```javascript
  var delta = newRotation - currentRotation;
  if (delta > 180) delta -= 360;
  if (delta < -180) delta += 360;
  currentRotation += delta;
  ```
  Wait, simpler: just track `currentRotation = angle - startAngle` but handle the atan2 wrap. Actually, the standard approach from RESEARCH.md is to compute the cumulative delta correctly. The key insight: store `lastAngle` and compute per-move delta:
  ```javascript
  var angle = getAngle(e);
  var delta = angle - lastAngle;
  if (delta > 180) delta -= 360;
  if (delta < -180) delta += 360;
  lastAngle = angle;
  currentRotation += delta;
  ```
  Initialize `lastAngle` in pointerdown: `lastAngle = getAngle(e);`
- Apply: `groupEl.setAttribute('transform', 'rotate(' + currentRotation + ', 240, 240)')`
- Sample velocity: push {angle: currentRotation, time: performance.now()}, keep last 3
- Accumulate dragDistance: `dragDistance += Math.abs(delta)`

`pointerup` on svgEl:
- if (!isDragging) return
- isDragging = false

**e) Inertia and snap** — per RESEARCH.md Pattern 3:
```javascript
function applyInertiaAndSnap() {
  if (reducedMotion || velocitySamples.length < 2) { snapToNearest(); return; }
  var first = velocitySamples[0];
  var last = velocitySamples[velocitySamples.length - 1];
  var dt = last.time - first.time;
  var velocity = dt > 0 ? (last.angle - first.angle) / dt : 0;
  var DECAY = 0.92;
  var FRAME_MS = 16.67;
  var MAX_OVERSHOOT = 30;
  function inertiaFrame() {
    velocity *= DECAY;
    currentRotation += velocity * FRAME_MS;
    var nearest = Math.round(currentRotation / SEGMENT_ANGLE) * SEGMENT_ANGLE;
    if (Math.abs(currentRotation - nearest) > MAX_OVERSHOOT) {
      currentRotation = nearest + (currentRotation > nearest ? MAX_OVERSHOOT : -MAX_OVERSHOOT);
      velocity = 0;
    }
    groupEl.setAttribute('transform', 'rotate(' + currentRotation + ', 240, 240)');
    if (Math.abs(velocity) > 0.05) {
      rafId = requestAnimationFrame(inertiaFrame);
    } else {
      snapToNearest();
    }
  }
  rafId = requestAnimationFrame(inertiaFrame);
}
```

```javascript
function snapToNearest() {
  var nearest = Math.round(currentRotation / SEGMENT_ANGLE) * SEGMENT_ANGLE;
  currentRotation = nearest;
  var idx = ((Math.round(-nearest / SEGMENT_ANGLE) % 5) + 5) % 5;
  if (!reducedMotion) {
    groupEl.classList.add('wheel-snap-transition');
    groupEl.style.transform = 'rotate(' + nearest + 'deg)';
    setTimeout(function() {
      groupEl.classList.remove('wheel-snap-transition');
      groupEl.removeAttribute('style');
      groupEl.setAttribute('transform', 'rotate(' + nearest + ', 240, 240)');
    }, 250);
  } else {
    groupEl.setAttribute('transform', 'rotate(' + nearest + ', 240, 240)');
  }
  updateActive(idx);
}
```

```javascript
function rotateTo(idx) {
  if (rafId) cancelAnimationFrame(rafId);
  var targetRotation = -idx * SEGMENT_ANGLE;
  // Shortest path normalization
  var delta = targetRotation - currentRotation;
  while (delta > 180) delta -= 360;
  while (delta < -180) delta += 360;
  currentRotation = currentRotation + delta;
  if (!reducedMotion) {
    groupEl.classList.add('wheel-snap-transition');
    groupEl.style.transform = 'rotate(' + currentRotation + 'deg)';
    setTimeout(function() {
      groupEl.classList.remove('wheel-snap-transition');
      groupEl.removeAttribute('style');

    }, 250);
  } else {
    groupEl.setAttribute('transform', 'rotate(' + currentRotation + ', 240, 240)');

}
```

```javascript
  // Update button aria-pressed

    btn.setAttribute('aria-pressed', i === idx ? 'true' : 'false');
  });
  // Update center content with crossfade
  var f = features[idx];
  if (reducedMotion) {
    setCenterContent(f);
  } else {
    centerContent.style.opacity = '0';
    setTimeout(function() {
      setCenterContent(f);
      centerContent.style.opacity = '1';
    }, 75);
  }
}

function setCenterContent(f) {
  var iconEl = centerContent.querySelector('svg');
  if (iconEl) iconEl.replaceChildren(); // clear existing
  if (iconEl && f.iconSvg) {
    // Create a temporary container to parse the static SVG content
    var temp = document.createElementNS('http://www.w3.org/2000/svg', 'svg');
    temp.innerHTML = f.iconSvg;
    while (temp.firstChild) { iconEl.appendChild(temp.firstChild); }
  }
  var titleEl = centerContent.querySelector('.wheel-center-title');
  if (titleEl) titleEl.textContent = f.title;
  var descEl = centerContent.querySelector('.wheel-center-desc');
  if (descEl) descEl.textContent = f.desc;
  var linkEl = centerContent.querySelector('.wheel-center-link');
  if (linkEl) {
    linkEl.href = f.href;
    linkEl.setAttribute('aria-label', 'Explore ' + f.title);
  }
}
```
