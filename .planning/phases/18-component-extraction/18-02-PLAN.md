**9. Add dropdown JS for language switcher** after the existing Features dropdown `<script is:inline>` block (after line 192). Add TWO new script blocks:

```html
<!-- Language switcher dropdown toggle (desktop) -->
<script is:inline>
  (function() {
    var dropdown = document.getElementById('lang-dropdown');
    if (!dropdown) return;
    var btn = dropdown.querySelector('button');
    var menu = dropdown.querySelector('.nav-dropdown');
    var chevron = btn.querySelector('.nav-chevron');

    function open() {
      menu.classList.add('is-open');
      btn.setAttribute('aria-expanded', 'true');
      if (chevron) chevron.style.transform = 'rotate(180deg)';
    }

    function close() {
      menu.classList.remove('is-open');
      btn.setAttribute('aria-expanded', 'false');
      if (chevron) chevron.style.transform = '';
    }

    btn.addEventListener('click', function(e) {
      e.stopPropagation();
      menu.classList.contains('is-open') ? close() : open();
    });

    var closeTimer = null;
    dropdown.addEventListener('mouseenter', function() { clearTimeout(closeTimer); open(); });
    dropdown.addEventListener('mouseleave', function() { closeTimer = setTimeout(close, 200); });
    document.addEventListener('click', function(e) { if (!dropdown.contains(e.target)) close(); });
    document.addEventListener('keydown', function(e) {
      if (e.key === 'Escape' && menu.classList.contains('is-open')) { close(); btn.focus(); }
    });
  })();
</script>

<!-- Language switcher dropdown toggle (mobile) -->
<script is:inline>
  (function() {
    var dropdown = document.getElementById('lang-dropdown-mobile');
    if (!dropdown) return;
    var btn = dropdown.querySelector('button');
```
