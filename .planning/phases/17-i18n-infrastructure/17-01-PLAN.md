4. Create `src/i18n/utils.ts` — translation utility (per D-02 fallback, D-03 nested objects):

```typescript
import enCommon from './locales/en/common';
import zhCommon from './locales/zh/common';
import frCommon from './locales/fr/common';

export type Locale = 'en' | 'zh' | 'fr';
export type CommonTranslations = typeof enCommon;

const translations: Record<Locale, CommonTranslations> = {
  en: enCommon,
  zh: zhCommon,
  fr: frCommon,
};

export function useTranslations(locale: Locale | undefined): CommonTranslations {
  return translations[locale ?? 'en'] ?? translations['en'];
}
```

Uses static imports (not dynamic `await import()`) matching the `blogPosts.ts` utility pattern. English baseline is the type source via `typeof enCommon`. Missing/undefined locale falls back to English per D-02.
